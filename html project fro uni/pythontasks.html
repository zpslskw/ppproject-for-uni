Примеры Python-заданий

n = int(input("Введите число: "))
for i in range(1, 11):
    print(f"{n} x {i} = {n * i}")

s = input("Введите строку: ").replace(" ", "").lower()
if s == s[::-1]:
    print("Это палиндром.")
else:
    print("Это не палиндром.")

nums = [12, 45, 23, 67, 45, 89, 89]
unique_nums = list(set(nums))
unique_nums.sort(reverse=True)
if len(unique_nums) >= 2:
    print("Второй максимум =", unique_nums[1])
else:
    print("Недостаточно уникальных элементов")

s = input("Введите строку: ")
count = sum(1 for ch in s if ch.isdigit())
print("Цифр в строке:", count)

def mean(numbers):
    return sum(numbers) / len(numbers) if numbers else 0

nums = [4, 8, 15, 16, 23, 42]
print("Среднее значение:", mean(nums))
